# Команды для запуска сервисов из командной строки
* Запуск Django
python manage.py runserver

* Worker (обработка задач)
celery -A config worker --loglevel=info --pool=solo --concurrency=10

* Beat (планировщик задач)
celery -A config beat --loglevel=info

* Flower (мониторинг)
celery -A config flower --port=5555

# Redis (в WSL)
* Общие
redis-server - Запуск с дефолтным конфигом
redis-server /etc/redis/redis.conf - Запуск с конфигурационным файлом
redis-cli ping - Проверка доступности

* Управление службой
sudo service redis-server start - Старт
sudo service redis-server stop - Стоп
sudo service redis-server restart - Перезапуск
sudo service redis-server status - Статус

* Работа с Redis через CLI (может понадобиться для просмотра данных о задачах)
redis-cli - Подключиться
info - Общая информация
info keyspace - Информация о ключах
KEYS * - Список всех ключей
KEYS prefix:* - Ключи с определенным префиксом
GET <key> - Получить значение по ключу

# Docker Compose
* Общие
docker-compose up --build о Собрать все образы и запустить
docker-compose up -d - Запустить в фоне
docker-compose down - Остановить все
docker-compose restart - перезапуск всего
docker-compose up --build web - Запустить конкретный сервис
docker-compose stop web - Остановить конкретный сервис

docker-compose down -v - Удалить все контейнеры

docker-compose logs - Все логи
docker-compose logs -f <имя контейнера> - Логи конкретного сервиса

docker-compose config - показать текущую конфигурацию

* Выполнение команд внутри контейнера
docker-compose exec web bash - войти в shell контейнера, тогда команды обычные
docker-compose exec <имя контейнера> <команда> - выполнить команду без входа, просто передать
